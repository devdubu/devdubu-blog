```java
/**
* /api/user/data
*/
@ApiOperation(value = "getUserInitialData() : nuxt store에서 사용되는 사용자 기본정보", notes = "사용자 정보(user) - 즐겨찾기 / 메뉴(menu) / 접근가능한 프로그램 목록(prgm) / 다국어(lang)")
@PostMapping(value = "/data")

public CommonResult getUserInitialData( @UserSession UserVo sessUser ) throws Exception {
		LogUtil.printRequestLog(request);
		Map<String,Object> env = new HashMap<>();
		env.put("divisionUse", envProp.getDIVISION_USE_YN());
		env.put("multilangUse", envProp.getMULTI_LANG_USE_YN());
		env.put("jwtUse", envProp.getJWT_USE_YN());
		env.put("singleauthUse", envProp.getSINGLE_AUTH_USE_YN());
		
		Map<String,Object> data = new HashMap<String, Object>();
		data.put("menu", ObjectUtil.addTreeObject(loginService.getUserMenuList(sessUser)));
		
		// 기존 Tree 구조로 되어있는 객체를 Array 구조로 보내주시면 됩니다.
		// 해당 부분에 menuList 라는 객체를 추가해주시면 됩니다.
		// 쿼리 부분은 loginService.getUserMenuList 를 참조하셔서 만들어주시면 됩니다.
		
		data.put("prgm", userService.getUserPrgmList(sessUser));
		data.put("lang", langService.getLangList(null));
		data.put("favoriteMenu", favMenuService.getFavoritesMenu());
		data.put("user", sessUser);
		data.put("env", env);
		return CommonResult.ofSuccess(data);
	}
```
