# HChart Interface 정의 완료

```tsx
interface IHBarChart {
    type: string | null
    name: string | null
    title?: string | HChartTitle | null
    subtitle?: string | HChartSubTitle | null
    options?: string | HOption | null
    xAxis?: HXAxis | null
    yAxis?: HXAxis | null
    series: HBarChartSeries[] | HLineChartSeries[] | HPieChartSeries[]
}

interface HChartTitle {
    text?: string | null
    gap?: number | null
    backgroundStyle?: {
        fill?: string | null
        padding?: string | null
        rx?: string | null
    }
    style?: {
        fill?: string | null
        fontSize?: string
    }
}

interface HChartSubTitle {
    text?: string | null
    style?: {
        fill: string | null
        fontSize?: string | null
        fontWeight?: string | null
        marginBottom?: string | null
    }
}

interface HOption {
    palette?: string | null
}

interface HXAxis {
    title?: string | null
    categories: Array<string> | null
    grid?: {
        visible?: boolean | null
    }
    label?: {
        step?: number | null
    }
}

interface HXAxis {
    title?: string | null
    tick?: boolean | null
}

interface HBarChartSeries {
    pointLabel?: {
        visible: boolean | null
        position: string | null
        effect: string | null
    }
    pointWidth?: number | null
    data?: any[]
}

interface HLineChartSeries {
    type: string | null
    lineType: string | null
    marker: {
        visible?: boolean
    }
    pointLabel?: boolean
    data?: any[]
}

interface HPieChartSeries {
    type?: string | null
    radius?: string | null
    legendByPoint?: boolean | null
    pointLabel: {
        text?: string | null
        visible?: boolean | null
        numberFormat?: string | null
        style: {
            fill?: string | null
            stroke?: string | null
            strokeWidth?: string | null
            fontSize?: string | null
        }
    }
    data?: any[]
}
```
