import { describe, it, expect } from 'vitest'
import type { IGridColumn } from '../type'
import { transforRealGridFields } from '.'
import { z } from 'zod'

const FeildTestType = z.object({
    fieldName: z.string(),
    dataType: z.union([z.literal('text'), z.literal('number')]),
})
describe('---- realGrid Field Setting 로직 테스트 ----', async () => {
    const columnConfig: IGridColumn[] = [
        { text: '회사', field: 'COMP_NM', width: 100, alignment: 'left' },
        { text: '코드 그룹CD', field: 'CODE_GRP_CD', width: 100, alignment: 'left' },
        { text: '코드 그룹명', field: 'CODE_GRP_NM', width: 150, alignment: 'left' },
        { text: '코드CD', field: 'CODE_CD', width: 120, alignment: 'left' },
        { text: '코드명', field: 'CODE_NM', width: 200, alignment: 'left' },
        { text: '언어', field: 'LANG', width: 100 },
        { text: '순서', field: 'CODE_ORD', width: 50 },
        { text: '사용', field: 'USE_FL', width: 50 },
        { text: '시스템코드', field: 'ADMIN_FL', width: 80 },
        { text: '최종수정시간', field: 'UPDATE_DT', width: 150 },
    ]

    const FieldArraySchema = z.array(FeildTestType)

    it('---- field 셋팅 출력 테스트 ----', () => {
        const result = transforRealGridFields(columnConfig)
        const parseResult = FieldArraySchema.safeParse(result)
        
        expect(parseResult.success).toBe(true)
    })

    it('---- column 셋팅 출력 테스트 ----', () => {})

})
