<template>

<!-- <add-user-dialog /> -->

<CDialog v-model:visible="openUserDialog" title="사용자 조회 / 선택 POPUP" :params="dialogParams" pcd="N20202" @close="closeUserDialog" />

<CDialog

v-model:visible="openUserDialogNoCode"

title="사용자 조회 / 선택 POPUP"

:params="noCodeDialogParams"

pcd="N20202"

@close="closeNoCodeUserDialog" />
</template>
<script>
  

const defaultCode = ref({

name: '관리자',

value: 'systemstd',

})

  

const noFieldCode = ref({

name: '관리자',

value: 'systemstd',

})

  

// *********************************************

// ****************** Dialog *******************

// *********************************************

  

const openUserDialog = ref(false)

  

const dialogParams = ref({

keyword: '',

})

  

const openUserDialogFunc = async () => {

openUserDialog.value = true

}

  

const closeUserDialog = (value) => {

if (value.result && value.data?.name && value.data?.value) {

defaultCode.value.name = value.data?.name

defaultCode.value.value = value.data?.value

}

}

  

const openUserDialogNoCode = ref(false)

  

const noCodeDialogParams = ref({

keyword: '',

})

  

const openNoCodeUserDialogFunc = () => {

openUserDialogNoCode.value = true

}

  

const closeNoCodeUserDialog = (value) => {

if (value.result && value.data?.name && value.data?.value) {

noFieldCode.value.name = value.data?.name

noFieldCode.value.value = value.data?.value

}

}

  

// *********************************************

// ****************** Event ********************

// *********************************************

  

const codeboxEvent = {

keyup: async (value) => {

dialogParams.value.keyword = value

openUserDialogFunc()

},

search: () => {

dialogParams.value.keyword = ''

openUserDialogFunc()

},

}

  

const noCodeboxEvent = {

keyup: async (value) => {

noCodeDialogParams.value.keyword = value

openNoCodeUserDialogFunc()

},

search: () => {

noCodeDialogParams.value.keyword = ''

openNoCodeUserDialogFunc()

},

}
</script>
