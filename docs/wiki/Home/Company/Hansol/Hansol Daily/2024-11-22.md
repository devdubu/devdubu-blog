- Master 와 childern 분리
- master에서 받고 childern으로 토스 하는 형태

```tsx
getChainSelectConfig (){
	const chainSelectResponse = { }
	// 해당 부분에서는 data 없는 것 분기 처리
	if(!editor.data) return chainSelectResponse 
	
	if(child) chainSelectResponse = getChainSelectChildConfig ()
	// 이후 MasterChainSelect로직 진행
	else chainSelectResponse = getChainSelectMasterConfig()

	return chainSelectResponse
}
```
